# 例题 洛谷P3807

题意：给定a,b,pa,b,pa,b,p，求C(a+b,b)modp(p≤105)C(a+b,b)\mod p\quad (p\le10^5)C(a+b,b)modp(p≤105)

#include <bits/stdc++.h>
#define ll long long

using namespace std;

ll mod;
ll inv(ll base){
    ll ret = 1, t = mod - 2;
    for(;t; t>>=1){
        if(t & 1) ret = ret * base % mod;
        base = base * base % mod;
    }
    return ret;
}
// 暴力求组合数，有优化空间
ll Comb(ll n, ll m){
    if(n < m) return 0;
    if(n - m < m) m = n - m;
    ll denom = 1, numer = 1;
    for(int i=0; i<m; i++){
        numer = numer * (n - i) % mod;
        denom = denom * (i + 1) % mod;
    }
    return numer * inv(denom) % mod;
}

ll Lucas(ll n, ll m){
    if(m == 0) return 1;
    return Lucas(n / mod, m / mod) * Comb(n % mod, m % mod) % mod;
}

signed main(){
    int t; cin >> t;
    while(t--) {
        ll a, b;
        cin >> a >> b >> mod;
        cout << Lucas(a + b, a) << '\n';
    }
}
