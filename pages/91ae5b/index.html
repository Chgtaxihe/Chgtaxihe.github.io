# 万能欧几里得

# AC代码

# 51Nod 1187

#include <bits/stdc++.h>
#define Android ios::sync_with_stdio(false), cin.tie(NULL)

using namespace std;

void calc(int a, int b, int c, int d, int & p, int & q){
    if(a == 0){
        p = 1, q = d / c + 1;
    } else if(a >= b){
        int delta = a / b;
        calc(a%b, b, c - d*delta, d, p, q);
        p += q * delta;
    }else if(c > d){
        p = q = 1;
    }else{
        calc(d, c, b, a, q, p);
    }
}

void solve(){
    int a, b, c, d, p, q, g;
    cin >> a >> b >> c >> d;
    calc(a, b, c, d, p, q);
    g = __gcd(p, q);
    cout << p / g << "/" << q / g << "\n";
}

signed main(){
    Android;
    int t; cin >> t;
    while(t--) solve();
}
